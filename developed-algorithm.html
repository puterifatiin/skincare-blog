Step 1: Data Preparation

Generate synthetic data for skin condition analysis.

Dataset Skin Analysis

skin_csv.py (click here)

import cv2
import pandas as pd
import numpy as np
import os

# Create a directory for sample images
os.makedirs('sample_images', exist_ok=True)

# Generate sample images and save them
for i in range(1000):
    img = np.random.randint(0, 255, (50, 50), dtype=np.uint8)
    cv2.imwrite(f'sample_images/img_{i}.png', img)

# Generate random data for the dataset
data = {
    'image_path': [f'sample_images/img_{i}.png' for i in range(1000)],
    'hydration': np.random.randint(0, 101, size=1000),
    'wrinkles': np.random.randint(0, 2, size=1000),
    'acne': np.random.randint(0, 2, size=1000),
    'pores_size': np.random.rand(1000) * 10,
    'pigmentation': np.random.randint(0, 2, size=1000),
    'redness': np.random.randint(0, 101, size=1000),
    'uniformness': np.random.randint(0, 101, size=1000),
    'eye_area': np.random.rand(1000) * 10
}

# Create a DataFrame
df = pd.DataFrame(data)

# Save to CSV
df.to_csv('skin_analysis_dataset.csv', index=False)

Directory Creation:

  • The script ensures that a directory named sample_images exists. If it does not, the script creates it. This directory will store all the generated images.

Image Generation:

  • It generates 1000 grayscale images, each of size 50x50 pixels, with random pixel values. These images are saved in the sample_images directory with filenames img_0.png, img_1.png, ..., img_999.png.

Data Generation:

  • It generates random values for skin attributes (hydration, wrinkles, acne, pores_size, pigmentation, redness, uniformness, eye_area) for 1000 samples.

DataFrame Creation:

  • It constructs a Pandas DataFrame (df) using the generated data and links each sample image path to its corresponding attribute values.

CSV Export:

  • It saves the DataFrame to a CSV file (skin_analysis_dataset.csv) for later use in training and testing machine learning models.

Step 2: Model Training and Prediction

Use two models to predict skin condition percentages based on facial images through webcam.

Support Vector Regression (SVR)

skin_svr.py (click here)

import cv2
import numpy as np
import pandas as pd
from sklearn.preprocessing import StandardScaler
from sklearn.svm import SVR
from sklearn.multioutput import MultiOutputRegressor
from sklearn.model_selection import train_test_split

# Initialize camera
cap = cv2.VideoCapture(0)

# Load pre-trained face detection model
face_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_frontalface_default.xml')

# Initialize StandardScaler
scaler = StandardScaler()

# Create SVR model wrapped with MultiOutputRegressor
svr_model = MultiOutputRegressor(SVR(kernel='rbf', C=100, gamma=0.1, epsilon=0.1))

# Function to load images from dataset
def load_images(df):
    images = []
    for img_path in df['image_path']:
        img = cv2.imread(img_path, cv2.IMREAD_GRAYSCALE)
        img = cv2.resize(img, (50, 50))  # Resize to 50x50 pixels
        images.append(img)
    return np.array(images)

# Function to fit the scaler and SVR model
def fit_svr_model(train_images, train_labels):
    # Flatten images for SVR
    def flatten_images(images):
        return images.reshape(images.shape[0], -1)

    train_images_flat = flatten_images(train_images)

    # Standardize features
    train_images_flat_scaled = scaler.fit_transform(train_images_flat)

    # Fit the SVR model
    svr_model.fit(train_images_flat_scaled, train_labels)

# Function to recognize faces and predict using SVR
def recognize_face_with_svr(frame):
    gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
    faces = face_cascade.detectMultiScale(gray, scaleFactor=1.1, minNeighbors=5, minSize=(30, 30))

    for (x, y, w, h) in faces:
        # Extract face region
        face = gray[y:y+h, x:x+w]
        face_resized = cv2.resize(face, (50, 50))  # Resize to 50x50 pixels
        face_resized = face_resized.astype('float32') / 255.0
        face_flat = face_resized.flatten().reshape(1, -1)

        # Check if scaler is fitted
        if hasattr(scaler, 'mean_'):
            # Transform face data
            face_flat = scaler.transform(face_flat)

            # Predict skin conditions using SVR model
            predictions = svr_model.predict(face_flat)
            hydration, wrinkles, acne, pores_size, pigmentation, redness, uniformness, eye_area = predictions[0]

            # Convert predictions to percentages
            total = hydration + wrinkles + abs(acne) + pores_size + pigmentation + redness + uniformness + eye_area
            hydration_percent = (hydration / total) * 100
            wrinkles_percent = (wrinkles / total) * 100
            acne_percent = (abs(acne) / total) * 100
            pores_percent = (pores_size / total) * 100
            pigmentation_percent = (pigmentation / total) * 100
            redness_percent = (redness / total) * 100
            uniformness_percent = (uniformness / total) * 100
            eye_area_percent = (eye_area / total) * 100

            # Draw rectangle around face and display predicted labels
            cv2.rectangle(frame, (x, y), (x+w, y+h), (255, 0, 0), 2)
            cv2.putText(frame, f'SVR Hydration: {hydration_percent:.2f}%', (x, y-70), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
            cv2.putText(frame, f'SVR Wrinkles: {wrinkles_percent:.2f}%', (x, y-55), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
            cv2.putText(frame, f'SVR Acne: {acne_percent:.2f}%', (x, y-40), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
            cv2.putText(frame, f'SVR Pores: {pores_percent:.2f}%', (x, y-25), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
            cv2.putText(frame, f'SVR Pigmentation: {pigmentation_percent:.2f}%', (x, y-10), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
            cv2.putText(frame, f'SVR Redness: {redness_percent:.2f}%', (x, y+5), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
            cv2.putText(frame, f'SVR Uniformness: {uniformness_percent:.2f}%', (x, y+20), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
            cv2.putText(frame, f'SVR Eye Area: {eye_area_percent:.2f}%', (x, y+35), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)

    return frame

# Main workflow for SVR
def main_svr():
    # Load the dataset
    df = pd.read_csv('skin_analysis_dataset.csv')

    # Load images and labels
    images = load_images(df)
    labels = df[['hydration', 'wrinkles', 'acne', 'pores_size', 'pigmentation', 'redness', 'uniformness', 'eye_area']].values

    # Split the dataset into training and testing sets
    X_train, X_test, y_train, y_test = train_test_split(images, labels, test_size=0.2, random_state=42)

    # Fit the SVR model
    fit_svr_model(X_train, y_train)

    # Main loop
    while True:
        ret, frame = cap.read()
        if not ret:
            break

        # Recognize faces and predict skin conditions using SVR
        frame = recognize_face_with_svr(frame)

        # Display the resulting frame
        cv2.imshow('SVR Skin Condition Analysis', frame)

        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    # Release the capture
    cap.release()
    cv2.destroyAllWindows()

if __name__ == "__main__":
    main_svr()

# Create Random Forest Regressor model wrapped with MultiOutputRegressor with optimizations
rfr_model = MultiOutputRegressor(

Face Detection and Image Processing:

  • The code captures images from the webcam and processes them to detect faces using the Haar Cascade face detector. Each detected face is resized to a standard size (50x50 pixels) for consistent processing.

Feature Scaling:

  • The images are scaled using a StandardScaler which normalizes the image data. This step ensures that the input features are on a similar scale, which is important for the performance of both the SVR and RFR models.

Model Prediction:

  • The both models SVR and RFR make predictions based on the processed image of the user's face. These models have been trained on a dataset with various skin conditions, so they should be able to predict different skin conditions (hydration, wrinkles, acne, etc.) for different users.

Personalized Predictions:

  • Since the models are trained to predict skin conditions from images, and every user will have a different face with different skin characteristics, the predictions should be personalized. If a new user has different hydration levels, wrinkles, acne, etc., the model should output different percentages for each of these conditions compared to another user.
